Video configuration error: cloudflareStream.customerCode is missing from site config.

{{- else -}} {{- $baseSrc := printf "https://customer-%s.cloudflarestream.com/%s/iframe" $customerCode $id -}} {{- $params := slice -}} {{- /* Load site-wide video defaults (tier 2). Tier 1 = page frontmatter / shortcode, tier 3 = absent (CF applies its own default). False values in either tier map to CF's "absent" model — never emit param=false. */ -}} {{- $videoDefaults := dict -}} {{- with site.Params.cloudflareStream -}} {{- with .videoDefaults -}} {{- $videoDefaults = . -}} {{- end -}} {{- end -}} {{- /* Map lowercase keys (Hugo normalises frontmatter) to CF embed querystring names. poster is handled separately via poster_image with URL construction (issue #45). */ -}} {{- $cfParamNames := dict "autoplay" "autoplay" "muted" "muted" "loop" "loop" "controls" "controls" "preload" "preload" "starttime" "startTime" "primarycolor" "primaryColor" "letterboxcolor" "letterboxColor" -}} {{- range $lowerKey, $cfKey := $cfParamNames -}} {{- /* Three-tier precedence: page frontmatter > site defaults > absent. isset distinguishes "explicitly set to false" from "not set at all". Only truthy values forwarded — false/empty omits the param entirely (CF model). */ -}} {{- if isset $video $lowerKey -}} {{- if index $video $lowerKey -}} {{- $params = $params | append (printf "%s=%v" $cfKey (index $video $lowerKey)) -}} {{- end -}} {{- else if isset $videoDefaults $lowerKey -}} {{- if index $videoDefaults $lowerKey -}} {{- $params = $params | append (printf "%s=%v" $cfKey (index $videoDefaults $lowerKey)) -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Resolve effective autoplay for the iframe loading attribute. loading="lazy" defers iframe initialisation until scroll, missing the CF player's autoplay window. Use loading="eager" when autoplay is effectively true. */ -}} {{- $effectiveAutoplay := false -}} {{- if isset $video "autoplay" -}} {{- $effectiveAutoplay = index $video "autoplay" -}} {{- else if isset $videoDefaults "autoplay" -}} {{- $effectiveAutoplay = index $videoDefaults "autoplay" -}} {{- end -}} {{- /* Inventory lookup — done once; used for both dimension resolution and poster_image fallback */ -}} {{- $invEntry := dict -}} {{- $inventory := index site.Data "video-inventory" -}} {{- if $inventory -}} {{- $matches := where $inventory.videos "uid" $id -}} {{- if gt (len $matches) 0 -}} {{- $invEntry = index $matches 0 -}} {{- end -}} {{- end -}} {{- /* Resolve aspect-ratio in priority order: 1) frontmatter width/height 2) data/video-inventory.yaml keyed by UID 3) Cloudflare Stream API (requires CLOUDFLARE_API_TOKEN + CLOUDFLARE_ACCOUNT_ID env vars) 4) 16:9 CSS default (no inline style) */ -}} {{- $width := $video.width -}} {{- $height := $video.height -}} {{- /* 2) Video inventory */ -}} {{- if and (not $width) (not $height) -}} {{- $width = $invEntry.width -}} {{- $height = $invEntry.height -}} {{- end -}} {{- /* 3) Cloudflare Stream API fallback Requires CLOUDFLARE_API_TOKEN and CLOUDFLARE_ACCOUNT_ID in env, with both vars listed under security.funcs.getenv in hugo.yaml. Skipped silently when either var is absent — regression tests rely on this (no creds set = no API call). */ -}} {{- if and (not $width) (not $height) -}} {{- $apiToken := os.Getenv "CLOUDFLARE_API_TOKEN" -}} {{- $accountId := os.Getenv "CLOUDFLARE_ACCOUNT_ID" -}} {{- if and $apiToken $accountId -}} {{- $apiUrl := printf "https://api.cloudflare.com/client/v4/accounts/%s/stream/%s" $accountId $id -}} {{- with resources.GetRemote $apiUrl (dict "headers" (dict "Authorization" (printf "Bearer %s" $apiToken))) -}} {{- $data := . | transform.Unmarshal -}} {{- if and $data $data.success -}} {{- $width = $data.result.input.width -}} {{- $height = $data.result.input.height -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Resolve poster_image in priority order: 1) frontmatter/shortcode poster_image 2) inventory poster_image (absolute URL stored at inventory level) 3) no poster — CF Stream default (first frame of video) URL construction rules: absolute URL (http/https prefix) → used verbatim site-root-relative (leading /) → prepend site baseURL relative filename (no leading /) → prepend page permalink */ -}} {{- $posterImage := $video.poster_image -}} {{- if not $posterImage -}} {{- $posterImage = $invEntry.poster_image -}} {{- end -}} {{- if $posterImage -}} {{- $posterURL := "" -}} {{- if or (hasPrefix $posterImage "http://") (hasPrefix $posterImage "https://") -}} {{- $posterURL = $posterImage -}} {{- else if hasPrefix $posterImage "/" -}} {{- $posterURL = printf "%s%s" (strings.TrimSuffix "/" site.BaseURL) $posterImage -}} {{- else -}} {{- $posterURL = printf "%s%s" $permalink $posterImage -}} {{- end -}} {{- $params = $params | append (printf "poster=%s" $posterURL) -}} {{- end -}} {{- $src := $baseSrc -}} {{- if gt (len $params) 0 -}} {{- $src = printf "%s?%s" $baseSrc (delimit $params "&") -}} {{- end -}} {{- $aspectStyle := "" -}} {{- if and $width $height -}} {{- $aspectStyle = printf "aspect-ratio: %v / %v;" $width $height -}} {{- end -}}
