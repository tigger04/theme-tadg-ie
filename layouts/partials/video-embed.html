{{- else -}} {{- $baseSrc := printf "https://customer-%s.cloudflarestream.com/%s/iframe" $customerCode $id -}} {{- $params := slice -}} {{- /* Map lowercase keys (Hugo normalises frontmatter) to CF embed querystring names */ -}} {{- $cfParamNames := dict "autoplay" "autoplay" "muted" "muted" "loop" "loop" "controls" "controls" "preload" "preload" "poster" "poster" "starttime" "startTime" "primarycolor" "primaryColor" "letterboxcolor" "letterboxColor" -}} {{- range $lowerKey, $cfKey := $cfParamNames -}} {{- if isset $video $lowerKey -}} {{- $val := index $video $lowerKey -}} {{- /* Only forward truthy values — false/empty means "use CF default (off)" */ -}} {{- if $val -}} {{- $params = $params | append (printf "%s=%v" $cfKey $val) -}} {{- end -}} {{- end -}} {{- end -}} {{- $src := $baseSrc -}} {{- if gt (len $params) 0 -}} {{- $src = printf "%s?%s" $baseSrc (delimit $params "&") -}} {{- end -}} {{- /* Resolve aspect-ratio in priority order: 1) frontmatter width/height 2) data/video-inventory.yaml keyed by UID 3) Cloudflare Stream API (requires CLOUDFLARE_API_TOKEN + CLOUDFLARE_ACCOUNT_ID env vars) 4) 16:9 CSS default (no inline style) */ -}} {{- $width := $video.width -}} {{- $height := $video.height -}} {{- /* 2) Video inventory lookup */ -}} {{- if and (not $width) (not $height) -}} {{- $inventory := index site.Data "video-inventory" -}} {{- if $inventory -}} {{- $matches := where $inventory.videos "uid" $id -}} {{- if gt (len $matches) 0 -}} {{- $entry := index $matches 0 -}} {{- $width = $entry.width -}} {{- $height = $entry.height -}} {{- end -}} {{- end -}} {{- end -}} {{- /* 3) Cloudflare Stream API fallback Requires CLOUDFLARE_API_TOKEN and CLOUDFLARE_ACCOUNT_ID in env, with both vars listed under security.funcs.getenv in hugo.yaml.
