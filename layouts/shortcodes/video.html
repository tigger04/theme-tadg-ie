{{- /* ABOUTME: Video shortcode supporting local files and Cloudflare Stream embeds */ -}} {{- /* ABOUTME: Auto-detects source type from argument: file path → HTML5 video, 32-char hex → CF Stream iframe */ -}} {{- $id := "" -}} {{- $isCFStream := false -}} {{- /* Detection: named "id" param takes priority, then positional arg with hex check */ -}} {{- if .Get "id" -}} {{- $id = .Get "id" -}} {{- $isCFStream = true -}} {{- else if .Get 0 -}} {{- if findRE "^[0-9a-f]{32}$" (.Get 0) -}} {{- $id = .Get 0 -}} {{- $isCFStream = true -}} {{- end -}} {{- end -}} {{- if $isCFStream -}} {{- /* Build video map for the shared partial */ -}} {{- $customerCode := site.Params.cloudflareStream.customerCode -}} {{- $videoMap := dict "id" $id -}} {{- /* Shortcode params preserve case; lowercase keys to match the partial's lookup */ -}} {{- $optionalParams := slice "autoplay" "muted" "loop" "controls" "preload" "poster" "startTime" "primaryColor" "letterboxColor" -}} {{- range $optionalParams -}} {{- $val := $.Get . -}} {{- if $val -}} {{- $videoMap = merge $videoMap (dict (lower .) $val) -}} {{- end -}} {{- end -}} {{- partial "video-embed.html" (dict "video" $videoMap "customerCode" $customerCode) -}} {{- else -}} {{- /* Local HTML5 video (original behaviour) */ -}} {{- end -}}
